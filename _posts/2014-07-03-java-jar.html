1. java运行jar包
打包以后的jar包依赖的其他jar包在lib目录下，需要修改CLASSPATH。使用脚本： main函数在com.mytest.test中 jar包为test.jar
#!/bin/sh LIBPATH=lib CLASSPATH=test.jar OSName=`uname -o` for file in ${LIBPATH}/*.jar; do case $OSName in *Cygwin* ) CLASSPATH="${CLASSPATH};$file" ;; * ) CLASSPATH="${CLASSPATH}:$file" ;; esac done export CLASSPATH java -classpath "${CLASSPATH}" com.mytest.test
